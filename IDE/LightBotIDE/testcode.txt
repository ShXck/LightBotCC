	// comment

	ChangeDir([RIGHT]);

        High Block;
	FEnd;

    Set r = 6;

    Pos(r,s);
	For k = 2 Times
	    Call proc2;
	FEnd;

	Set ver = 7;

    Call myProc;

	Var hor_robot = 1;
